126 opt_types = {
127 COAP_OPTION_URI_PATH: str,
128 COAP_OPTION_URI_HOST: str,
129 COAP_OPTION_LOCATION_PATH: str,
130 COAP_OPTION_URI_QUERY: str,

137 opts = {}
138 while True:
139 option = coap_option_next(ct.byref(opt_iter), llapi_check=False)
140 if not option:
141 break
142
143 if opt_iter.number in opt_types:
144 typ = opt_types[opt_iter.number]
145
146 if typ == str:
147 value = ct.string_at(coap_opt_value(option), coap_opt_length(option))
148 elif typ == bytes:
149 value = bytes(ct.cast(coap_opt_value(option), ct.POINTER(ct.c_char * coap_opt_length(option))))
150 elif opt_iter.number == COAP_OPTION_CONTENT_FORMAT:
151 value = coap_decode_var_bytes(coap_opt_value(option), coap_opt_length(option))
152 if lookup_names:
153 for key in dir(llapi):
154 if key.startswith("COAP_MEDIATYPE_"):
155 if getattr(llapi, key, False) == value:
156 value = key[len("COAP_MEDIATYPE_"):].lower()
157 break
158 else:
159 if coap_opt_length(option) <= 4:
160 value = coap_decode_var_bytes(coap_opt_value(option), coap_opt_length(option))
161 elif coap_opt_length(option) <= 8:
162 value = coap_decode_var_bytes8(coap_opt_value(option), coap_opt_length(option))
163 else:
164 value = bytes(ct.cast(coap_opt_value(option), ct.POINTER(ct.c_char * coap_opt_length(option))))
165
166 if lookup_names:

170 if key[len("COAP_OPTION_"):].lower() not in opts:
171 opts[key[len("COAP_OPTION_"):].lower().replace("_","-")] = []
172 opts[key[len("COAP_OPTION_"):].lower().replace("_","-")].append(value)
