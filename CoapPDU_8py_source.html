159 def getOptions(self, lookup_names=False):
160 from . import llapi
161
162 opt_iter = coap_opt_iterator_t()
163 coap_option_iterator_init(self.lcoap_pdu, ct.byref(opt_iter), COAP_OPT_ALL)
164
165 opt_types = {
166 COAP_OPTION_URI_PATH: str,
167 COAP_OPTION_URI_HOST: str,
168 COAP_OPTION_LOCATION_PATH: str,
169 COAP_OPTION_URI_QUERY: str,
170 COAP_OPTION_LOCATION_QUERY: str,
171 COAP_OPTION_PROXY_URI: str,
172 COAP_OPTION_PROXY_SCHEME: str,
173 COAP_OPTION_RTAG: bytes,
174 }
175
176 opts = {}
177 while True:
178 option = coap_option_next(ct.byref(opt_iter), llapi_check=False)
179 if not option:
180 break
181
182 if opt_iter.number in opt_types:
183 typ = opt_types[opt_iter.number]
184
185 if typ == str:
186 value = ct.string_at(coap_opt_value(option), coap_opt_length(option))
187 elif typ == bytes:
188 value = bytes(ct.cast(coap_opt_value(option), ct.POINTER(ct.c_char * coap_opt_length(option))))
189 elif opt_iter.number == COAP_OPTION_CONTENT_FORMAT:
190 value = coap_decode_var_bytes(coap_opt_value(option), coap_opt_length(option))
191 if lookup_names:
192 for key in dir(llapi):
193 if key.startswith("COAP_MEDIATYPE_"):
194 if getattr(llapi, key, False) == value:
195 value = key[len("COAP_MEDIATYPE_"):].lower()
196 break
197 else:
198 if coap_opt_length(option) <= 4:
199 value = coap_decode_var_bytes(coap_opt_value(option), coap_opt_length(option))
200 elif coap_opt_length(option) <= 8:
201 value = coap_decode_var_bytes8(coap_opt_value(option), coap_opt_length(option))
202 else:
203 value = bytes(ct.cast(coap_opt_value(option), ct.POINTER(ct.c_char * coap_opt_length(option))))
204
205 if lookup_names:
206 for key in dir(llapi):
207 if key.startswith("COAP_OPTION_"):
208 if getattr(llapi, key, False) == opt_iter.number:
209 if key[len("COAP_OPTION_"):].lower() not in opts:
210 opts[key[len("COAP_OPTION_"):].lower().replace("_","-")] = []
211 opts[key[len("COAP_OPTION_"):].lower().replace("_","-")].append(value)
212 else:
213 opts[opt_iter.number] = value
214
215 return opts
